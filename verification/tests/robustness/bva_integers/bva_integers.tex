% BVA Test for TeX Integers
\count0=2147483647
\immediate\write16{Count 0 is \the\count0}

% This should trigger "Number too big"
\count1=2147483648
\immediate\write16{Count 1 is \the\count1}

\count2=-2147483647
\immediate\write16{Count 2 is \the\count2}

% This should trigger "Number too big" and result in -2147483647
\count3=-2147483648
\immediate\write16{Count 3 is \the\count3}

\count4=0
\immediate\write16{Count 4 is \the\count4}

\bye
